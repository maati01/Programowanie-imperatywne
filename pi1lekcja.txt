#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

bool is_prime(int num)
{
    int ii = 2;
    if (num <= 1)
    {
        return false;
    }
    while(ii*ii<=num)
    {
        if (num%ii==0)
        {
            return false;
        }
        ii++;
    }
    return true;
}

int main()
{
    int liczba = 0;
    printf("Podaj liczbe: ");
    scanf("%d", &liczba);
    printf("Liczba %d ",liczba);
    if (is_prime(liczba))
    {
        printf("jest liczba pierwsza.");
    }
    else
    {
        printf("nie jest liczba pierwsza.");
    }
    return 0;
}
